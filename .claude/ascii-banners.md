## 🛠️ **Advanced Generation Scripts**

### Script 1: Dynamic Color Banner Generator
```bash
#!/bin/bash
# figlet-color-banner.sh
# Usage: ./figlet-color-banner.sh "TEXT" color_code

text="${1:-podAI}"
color="${2:-93}"  # Default violet

# Generate with figlet and apply color
figlet -f big "$text" | while IFS= read -r line; do
    echo -e "\033[38;5;${color}m${line}\033[0m"
done
```

### Script 2: Rainbow Gradient Generator
```bash
#!/bin/bash
# rainbow-banner.sh
# Creates rainbow gradient effect

text="${1:-Prompt or Die}"
colors=(196 202 208 214 220 226 190 154 118 82 46 47 48 49 50 51)

echo "$text" | figlet -f slant | while IFS= read -r line; do
    color_index=0
    for (( i=0; i<${#line}; i++ )); do
        char="${line:$i:1}"
        if [[ "$char" != " " ]]; then
            color=${colors[$((color_index % ${#colors[@]}))]}
            echo -en "\033[38;5;${color}m${char}\033[0m"
            ((color_index++))
        else
            echo -n "$char"
        fi
    done
    echo
done
```

### Script 3: Matrix-Style Falling Code
```bash
#!/bin/bash
# matrix-banner.sh
# Creates falling matrix effect with banner

banner_text="${1:-podAI}"

# Generate base banner
base_banner=$(figlet -f big "$banner_text")

# Add matrix effect
echo -e "\033[48;5;232m"  # Black background
echo -e "\033[38;5;46m"    # Green text

# Animated falling characters effect (simplified)
for i in {1..3}; do
    echo -e "\033[2J\033[H"  # Clear screen and home cursor
    
    # Random matrix characters
    for j in {1..5}; do
        printf "%.s \033[38;5;46m%s\033[0m" $(seq 1 $((RANDOM%80))) \
               "$(echo "ﾊﾐﾋｰｳｼﾅﾓﾆﾇ" | cut -c$((RANDOM%8+1)))"
        echo
    done
    
    echo -e "\033[38;5;93m$base_banner\033[0m"
    
    # More falling characters
    for j in {1..3}; do
        printf "%.s \033[38;5;46m%s\033[0m" $(seq 1 $((RANDOM%80))) \
               "$(echo "ﾊﾐﾋｰｳｼﾅﾓﾆﾇ" | cut -c$((RANDOM%8+1)))"
        echo
    done
    
    sleep 0.5
done

echo -e "\033[0m"  # Reset
```
